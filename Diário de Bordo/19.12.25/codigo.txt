#include "esp_camera.h"

// ================== DEFINES ==================
#define FAIXA 10   // altura da faixa horizontal

#define PWDN_GPIO_NUM     32
#define RESET_GPIO_NUM    -1
#define XCLK_GPIO_NUM      0
#define SIOD_GPIO_NUM     26
#define SIOC_GPIO_NUM     27
#define Y9_GPIO_NUM       35
#define Y8_GPIO_NUM       34
#define Y7_GPIO_NUM       39
#define Y6_GPIO_NUM       36
#define Y5_GPIO_NUM       21
#define Y4_GPIO_NUM       19
#define Y3_GPIO_NUM       18
#define Y2_GPIO_NUM        5
#define VSYNC_GPIO_NUM    25
#define HREF_GPIO_NUM     23
#define PCLK_GPIO_NUM      2

// ================== RGB → HSV ==================
void rgb2hsv(uint8_t r, uint8_t g, uint8_t b,
             float &h, float &s, float &v) {


  float rf = r / 255.0;
  float gf = g / 255.0;
  float bf = b / 255.0;

  float maxv = max(rf, max(gf, bf));
  float minv = min(rf, min(gf, bf));
  float delta = maxv - minv;

  v = maxv;

  if (delta < 0.00001) {
    h = 0;
    s = 0;
    return;
  }

  s = (maxv == 0) ? 0 : (delta / maxv);

  if (maxv == rf)
    h = 60 * fmod(((gf - bf) / delta), 6);
  else if (maxv == gf)
    h = 60 * (((bf - rf) / delta) + 2);
  else
    h = 60 * (((rf - gf) / delta) + 4);

  if (h < 0) h += 360;
}

// ================== DETECÇÃO DE COR ==================
String detectaCor(float H, float S, float V) {

  if (V < 0.15) return "Preto";
  if (V > 0.85 && S < 0.20) return "Branco";

  if (H < 15 || H > 330) return "Vermelho";
  if (H < 40)           return "Laranja";
  if (H < 70)           return "Amarelo";
  if (H < 170)          return "Verde";
  if (H < 200)          return "Ciano";
  if (H < 240)          return "Azul";
  if (H < 310)          return "Roxo";

  return "Indefinido";
}

// ================== SETUP ==================
void setup() {
  Serial.begin(115200);

  camera_config_t config;
  config.ledc_channel = LEDC_CHANNEL_0;
  config.ledc_timer   = LEDC_TIMER_0;
  config.pin_d0       = Y2_GPIO_NUM;
  config.pin_d1       = Y3_GPIO_NUM;
  config.pin_d2       = Y4_GPIO_NUM;
  config.pin_d3       = Y5_GPIO_NUM;
  config.pin_d4       = Y6_GPIO_NUM;
  config.pin_d5       = Y7_GPIO_NUM;
  config.pin_d6       = Y8_GPIO_NUM;
  config.pin_d7       = Y9_GPIO_NUM;
  config.pin_xclk     = XCLK_GPIO_NUM;
  config.pin_pclk     = PCLK_GPIO_NUM;
  config.pin_vsync    = VSYNC_GPIO_NUM;
  config.pin_href     = HREF_GPIO_NUM;
  config.pin_sscb_sda = SIOD_GPIO_NUM;
  config.pin_sscb_scl = SIOC_GPIO_NUM;
  config.pin_pwdn     = PWDN_GPIO_NUM;
  config.pin_reset    = RESET_GPIO_NUM;
  config.xclk_freq_hz = 20000000;
  config.pixel_format = PIXFORMAT_RGB565;
  config.frame_size   = FRAMESIZE_QQVGA; // 160x120
  config.fb_count     = 1;

  if (esp_camera_init(&config) != ESP_OK) {
    Serial.println("Falha ao iniciar câmera");
    while (true);
  }
}

// ================== LOOP ==================
void loop() {

  camera_fb_t *fb = esp_camera_fb_get();
  if (!fb) {
    Serial.println("Erro ao capturar frame");
    return;
  }

  int largura = fb->width;
  int altura  = fb->height;

  // ===== FAIXA HORIZONTAL CENTRAL =====
  int y_ini = (altura / 2) - (FAIXA / 2);
  int y_fim = y_ini + FAIXA;

  float somaH = 0;
  float somaS = 0;
  float somaV = 0;
  int cont = 0;
  int primeiro_preto = -1; // Adicionado o int para declarar a variável. Iniciar com -1, pois o 0 é um pixel válido pra espcam.
  int ultimo_preto = -1; // Mesma coisa de cima

  for (int y = y_ini; y < y_fim; y++) {
    for (int x = 0; x < largura; x++) {

      int index = 2 * (y * largura + x);
      uint16_t pixel = (fb->buf[index] << 8) | fb->buf[index + 1];

      uint8_t r = ((pixel >> 11) & 0x1F) * 255 / 31;
      uint8_t g = ((pixel >> 5)  & 0x3F) * 255 / 63;
      uint8_t b = ( pixel        & 0x1F) * 255 / 31;

      float H, S, V;
      rgb2hsv(r, g, b, H, S, V);

      if (detectaCor(H,S,V) == "Preto") {  // Faltou o parêntesis e o espaço no if
        if (primeiro_preto == -1) { //Alteração de 0 para -1
          primeiro_preto == x; //Começo da linha
        }
      ultimo_preto == x;
      }
      // Acumula os valores HSV de todos os pixels da faixa
      somaH += H;
      somaS += S;
      somaV += V;
      cont++;
    }
  }

  esp_camera_fb_return(fb);

  // Cor média da faixa
  float Hmedio = somaH / cont;
  float Smedio = somaS / cont;
  float Vmedio = somaV / cont;
  String corDominante = detectaCor(Hmedio, Smedio, Vmedio);
 
  Serial.print("Cor dominante na faixa: ");
  Serial.print(corDominante);
  Serial.print("  | H: ");
  Serial.print(Hmedio, 1);
  Serial.print("  S: ");
  Serial.print(Smedio, 2);
  Serial.print("  V: ");
  Serial.println(Vmedio, 2);

 // Calculo do centro da linha
  if (primeiro_preto != -1) {
    int centro_linha  = (primeiro_preto + ultimo_preto) / 2;
    int centro_imagem = largura / 2;  // Vai dar 80px pois a largura total é 160px
    int erro = centro_linha - centro_imagem;

    // - Direção do carro - 
    // erro negativo = linha está para esquerda
    // erro zero = linha no centro
    // erro positivo = linha está para direita
    if (erro < -15) {
      Serial.println("VIRAR ESQUERDA   | erro: " + String(erro));
    } else if (erro > 15) {
      Serial.println("VIRAR DIREITA    | erro: " + String(erro));
    } else {
      Serial.println("SEGUIR EM FRENTE | erro: " + String(erro));
    }
  } else {
    Serial.println("Linha não encontrada!");
  }

  delay(50);
}
